53  final class RollingHdrQuantiles implements Quantiles {

55      private static final double[] PREDEFINED_QUANTILES = { 0, 0.5, 0.75, 0.9, 0.95, 0.98, 0.99, 0.999, 1.0 };
56      private static final Collection<Double> MONITORED = Arrays.stream(PREDEFINED_QUANTILES).boxed()
57                                                                .collect(toImmutableList());
58  
59      private final long intervalBetweenResettingMillis;
60      private final long creationTimestamp;
61      private final ArchivedHistogram[] archive;
62      private final boolean historySupported;
63      private final LongSupplier clock;
64      private final DoubleHistogram temporarySnapshotHistogram;
65  
66      private final Phase left;
67      private final Phase right;
68      private final Phase[] phases;
69      private final AtomicReference<Phase> currentPhaseRef;
70  
71      private double[] snapshot;
72      private long lastSnapshotTimeMillis;
73  
74      RollingHdrQuantiles() {
75          this(() -> new DoubleRecorder(2), 5, Duration.ofMinutes(2).toMillis(), System::currentTimeMillis);
76      }
77  
78      RollingHdrQuantiles(Supplier<DoubleRecorder> recorderSupplier,
79                          int numberHistoryChunks, long intervalBetweenResettingMillis,
80                          LongSupplier clock) {
81  
82          this.intervalBetweenResettingMillis = intervalBetweenResettingMillis;
83          this.clock = clock;
84          creationTimestamp = clock.getAsLong();
85  
86          left = new Phase(recorderSupplier, creationTimestamp + intervalBetweenResettingMillis);
87          right = new Phase(recorderSupplier, Long.MAX_VALUE);
88          phases = new Phase[] { left, right };
89          currentPhaseRef = new AtomicReference<>(left);
90  
91          historySupported = numberHistoryChunks > 0;
92          if (historySupported) {
93              archive = new ArchivedHistogram[numberHistoryChunks];
94              for (int i = 0; i < numberHistoryChunks; i++) {
95                  final DoubleHistogram archivedHistogram = createNonConcurrentCopy(left.intervalHistogram);
96                  archive[i] = new ArchivedHistogram(archivedHistogram, Long.MIN_VALUE);
97              }
98          } else {
99              archive = null;
100         }
101 
102         temporarySnapshotHistogram = createNonConcurrentCopy(left.intervalHistogram);
103     }
104 
105     @Override
106     public void observe(double value) {
107         observe(value, 0);
108     }
109 
110     public void observe(double value, double expectedIntervalBetweenValueSamples) {
111         final long currentTimeMillis = clock.getAsLong();
112         final Phase currentPhase = currentPhaseRef.get();
113         if (currentTimeMillis < currentPhase.proposedInvalidationTimestamp) {
114             currentPhase.recorder.recordValueWithExpectedInterval(value, expectedIntervalBetweenValueSamples);
115             return;
116         }
117 
118         final Phase nextPhase = currentPhase == left ? right : left;
119         nextPhase.recorder.recordValueWithExpectedInterval(value, expectedIntervalBetweenValueSamples);
120 
121         if (!currentPhaseRef.compareAndSet(currentPhase, nextPhase)) {
122             // another writer achieved progress and must submit rotation task to backgroundExecutor
123             return;
124         }
125 
126         // Current thread is responsible to rotate phases.
127         rotate(currentTimeMillis, currentPhase, nextPhase);
128     }
129 
130     private synchronized void rotate(long currentTimeMillis, Phase currentPhase, Phase nextPhase) {
131         try {
132             currentPhase.intervalHistogram =
133                     currentPhase.recorder.getIntervalHistogram(currentPhase.intervalHistogram);
134             addSecondToFirst(currentPhase.totalsHistogram, currentPhase.intervalHistogram);
135             if (historySupported) {
136                 // move values from recorder to correspondent archived histogram
137                 final long currentPhaseNumber =
138                         (currentPhase.proposedInvalidationTimestamp - creationTimestamp) /
139                         intervalBetweenResettingMillis;
140                 final int correspondentArchiveIndex = (int) (currentPhaseNumber - 1) % archive.length;
141                 final ArchivedHistogram correspondentArchivedHistogram = archive[correspondentArchiveIndex];
142                 reset(correspondentArchivedHistogram.histogram);
143                 addSecondToFirst(correspondentArchivedHistogram.histogram, currentPhase.totalsHistogram);
144                 correspondentArchivedHistogram.proposedInvalidationTimestamp =
145                         currentPhase.proposedInvalidationTimestamp +
146                         archive.length * intervalBetweenResettingMillis;
147             }
148             reset(currentPhase.totalsHistogram);
149         } finally {
150             final long millisSinceCreation = currentTimeMillis - creationTimestamp;
151             final long intervalsSinceCreation = millisSinceCreation / intervalBetweenResettingMillis;
152             currentPhase.proposedInvalidationTimestamp = Long.MAX_VALUE;
153             nextPhase.proposedInvalidationTimestamp =
154                     creationTimestamp + (intervalsSinceCreation + 1) * intervalBetweenResettingMillis;
155         }
156     }
157 
158     @Override
159     public Double get(double percentile) {
160         final double[] snapshot = getSnapshot();
161         for (int i = 0; i < PREDEFINED_QUANTILES.length; i++) {
162             if (percentile <= PREDEFINED_QUANTILES[i]) {
163                 return snapshot[i];
164             }
165         }
166         return snapshot[PREDEFINED_QUANTILES.length - 1];
167     }
168 
169     @Override
170     public Collection<Double> monitored() {
171         return MONITORED;
172     }
173 
174     private synchronized double[] getSnapshot() {
175         final long currentTimeMillis = clock.getAsLong();
176         final double[] oldSnapshot = snapshot;
177         if (oldSnapshot != null && currentTimeMillis - lastSnapshotTimeMillis < 1000) {
178             return oldSnapshot;
179         }
180 
181         reset(temporarySnapshotHistogram);
182 
183         for (Phase phase : phases) {
184             if (phase.isNeedToBeReportedToSnapshot(currentTimeMillis)) {
185                 phase.intervalHistogram = phase.recorder.getIntervalHistogram(phase.intervalHistogram);
186                 addSecondToFirst(phase.totalsHistogram, phase.intervalHistogram);
187                 addSecondToFirst(temporarySnapshotHistogram, phase.totalsHistogram);
188             }
189         }
190         if (historySupported) {
191             for (ArchivedHistogram archivedHistogram : archive) {
192                 if (archivedHistogram.proposedInvalidationTimestamp > currentTimeMillis) {
193                     addSecondToFirst(temporarySnapshotHistogram, archivedHistogram.histogram);
194                 }
195             }
196         }
197 
198         final double[] snapshot = new double[PREDEFINED_QUANTILES.length];
199         for (int i = 0; i < PREDEFINED_QUANTILES.length; i++) {
200             final double percentile = PREDEFINED_QUANTILES[i] * 100.0;
201             snapshot[i] = temporarySnapshotHistogram.getValueAtPercentile(percentile);
202         }
203         this.snapshot = snapshot;
204         lastSnapshotTimeMillis = currentTimeMillis;
205         return snapshot;
206     }
207 
208     @VisibleForTesting
209     synchronized void discardCachedSnapshot() {
210         snapshot = null;
211     }
212 
213     @VisibleForTesting
214     int estimatedFootprintInBytes() {
215         // each histogram has equivalent pessimistic estimation
216         int oneHistogramPessimisticFootprint = temporarySnapshotHistogram.getEstimatedFootprintInBytes();
217 
218         // 4 - two recorders with two histogram
219         // 2 - two histogram for storing accumulated values from current phase
220         // 1 - temporary histogram used for snapshot extracting
221         return oneHistogramPessimisticFootprint * ((archive != null ? archive.length : 0) + 4 + 2 + 1);
222     }
223 
224     private static void reset(DoubleHistogram histogram) {
225         if (histogram.getTotalCount() > 0) {
226             histogram.reset();
227         }
228     }
229 
230     private static void addSecondToFirst(DoubleHistogram first, DoubleHistogram second) {
231         if (second.getTotalCount() > 0) {
232             first.add(second);
233         }
234     }
235 
236     private static DoubleHistogram createNonConcurrentCopy(DoubleHistogram source) {
237         final DoubleHistogram copy = new DoubleHistogram(
238                 source.getHighestToLowestValueRatio(),
239                 source.getNumberOfSignificantValueDigits());
240         copy.setAutoResize(source.isAutoResize());
241         return copy;
242     }
243 
244     private static String histogramToString(DoubleHistogram histogram) {
245         try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
246             final PrintStream writer = new PrintStream(baos);
247             histogram.outputPercentileDistribution(writer, 1.0);
248             return new String(baos.toByteArray());
249         } catch (IOException e) {
250             throw new IllegalStateException(e);
251         }
252     }
253 
254     private static String printArray(Object[] array, String elementName) {
255         final StringBuilder buf = new StringBuilder();
256         buf.append('{');
257         for (int i = 0; i < array.length; i++) {
258             buf.append('\n')
259                .append(elementName)
260                .append('[').append(i).append("]=")
261                .append(array[i]);
262         }
263         buf.append("\n}");
264         return buf.toString();
265     }

267     private static final class ArchivedHistogram {
268 
269         final DoubleHistogram histogram;
270         volatile long proposedInvalidationTimestamp;
271 
272         ArchivedHistogram(DoubleHistogram histogram, long proposedInvalidationTimestamp) {
273             this.histogram = histogram;
274             this.proposedInvalidationTimestamp = proposedInvalidationTimestamp;
275         }
276 
277         @Override
278         public String toString() {
279             return "ArchivedHistogram{" +
280                     "\n, proposedInvalidationTimestamp=" + proposedInvalidationTimestamp +
281                     "\n, histogram=" + histogramToString(histogram) +
282                     "\n}";
283         }
284     }
285 
286     private final class Phase {
287 
288         final DoubleRecorder recorder;
289         final DoubleHistogram totalsHistogram;
290         DoubleHistogram intervalHistogram;
291         volatile long proposedInvalidationTimestamp;
292 
293         Phase(Supplier<DoubleRecorder> recorderSupplier, long proposedInvalidationTimestamp) {
294             recorder = recorderSupplier.get();
295             intervalHistogram = recorder.getIntervalHistogram();
296             intervalHistogram.setAutoResize(true);
297             totalsHistogram = intervalHistogram.copy();
298             totalsHistogram.setAutoResize(true);
299             this.proposedInvalidationTimestamp = proposedInvalidationTimestamp;
300         }
301 
302         @Override
303         public String toString() {
304             return "Phase{" +
305                     "\n, proposedInvalidationTimestamp=" + proposedInvalidationTimestamp +
306                     "\n, totalsHistogram=" + (totalsHistogram != null ? histogramToString(totalsHistogram)
307                                                                       : "null") +
308                     "\n, intervalHistogram=" + histogramToString(intervalHistogram) +
309                     "\n}";
310         }
311 
312         boolean isNeedToBeReportedToSnapshot(long currentTimeMillis) {
313             final long proposedInvalidationTimestampLocal = proposedInvalidationTimestamp;
314             if (proposedInvalidationTimestampLocal > currentTimeMillis) {
315                 return true;
316             }
317             if (!historySupported) {
318                 return false;
319             }
320             final long correspondentChunkProposedInvalidationTimestamp =
321                     proposedInvalidationTimestampLocal + archive.length * intervalBetweenResettingMillis;
322             return correspondentChunkProposedInvalidationTimestamp > currentTimeMillis;
323         }
324     }
325 
326     @Override
327     public String toString() {
328         return "RollingHdrQuantiles{" +
329                 "\nintervalBetweenResettingMillis=" + intervalBetweenResettingMillis +
330                 ",\n creationTimestamp=" + creationTimestamp +
331                 (!historySupported ? "" : ",\n archive=" + printArray(archive, "chunk")) +
332                 ",\n left=" + left +
333                 ",\n right=" + right +
334                 ",\n currentPhase=" + (currentPhaseRef.get() == left ? "left" : "right") +
335                 ",\n temporarySnapshotHistogram=" + histogramToString(temporarySnapshotHistogram)  +
336                 '}';
337     }
